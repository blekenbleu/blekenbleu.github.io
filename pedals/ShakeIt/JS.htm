iirH:  high-pass IIR

Run once javascript code
root["low"] = 38;	// whatever initial value should be
const tc = 3;		// higher values increase sensitivity
function iirH(i) {
	i -= root["low"]; 
	root["low"] += i / tc;
	return i;
}

iirL:  low-pass IIR

Run once javascript code
root["low"] = 38;	// whatever initial value should be
const tc = 3;		// higher values decrease sensitivity
function iirL(i) {
	root["low"] += (i - root["low"]) / tc;
	return root["low"];
}
